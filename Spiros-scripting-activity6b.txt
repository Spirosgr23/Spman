#!/bin/bash

echo -n "enter a number for a:"; read a
echo -n " enter a number for b:"; read b


if [[ -n ${a//[0-99999999]/} ]]; then
 { echo "Input accepts only numbers" & exit 0; }


 elif [[ -n ${b//[0-99999999]/} ]]; then
{ echo "Input accepts only numbers" & exit 0; }
fi

if [[ "$a" -gt "$b" ]]; then
 echo "True"

 elif [[ "$a" -lt "$b" ]]; then
 echo "False"

 fi
